## Enforce Using a JSON Schema

```lms_code_snippet
  variants:
    "Non-streaming":
      language: typescript
      code: |
        const result = await model.respond("Tell me about The Hobbit.", {
          structured: {
            type: "json",
            jsonSchema: {
              type: "object",
              properties: {
                title: { type: "string" },
                author: { type: "string" },
                year: { type: "integer" },
              },
              required: ["title", "author", "year"],
            },
          },
        });

        const book = JSON.parse(result.content);
        console.info(book);
    Streaming:
      language: typescript
      code: |
        const prediction = model.respond("Tell me about The Hobbit.", {
          structured: {
            type: "json",
            jsonSchema: {
              type: "object",
              properties: {
                title: { type: "string" },
                author: { type: "string" },
                year: { type: "integer" },
              },
              required: ["title", "author", "year"],
            },
          },
        });

        for await (const { content } of prediction) {
          process.stdout.write(content);
        }
        console.info(); // Print a newline

        const result = await prediction.result();
        const book = JSON.parse(result.content);

        console.info("Parsed", book);
```
